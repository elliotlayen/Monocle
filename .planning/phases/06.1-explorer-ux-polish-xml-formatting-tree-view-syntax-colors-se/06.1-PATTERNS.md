**Imports pattern** (lines 1-4):

**Core pattern** (lines 6-20):
```typescript
// Pure function, DOMParser-based, graceful error fallback returning original input
export function parseXml(content: string): ParseResult {
  if (!content || !content.trim()) {
    return { document: null, error: "Empty content" };
  }

  const parser = new DOMParser();
  const doc = parser.parseFromString(content, "text/xml");

  const errorNode = doc.querySelector("parsererror");
  if (errorNode) {
    return { document: null, error: errorNode.textContent ?? "Parse error" };
  }

  return { document: doc, error: null };
}
```

**Key conventions to follow:**
- No external dependencies -- use browser DOMParser (already used in xml-parser.ts)
- Return original content on parse error (graceful fallback)
- Export a single public function (`formatXml`) and keep helpers private
- No class wrappers -- plain exported functions
